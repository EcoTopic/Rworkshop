Exclude Totals

malaria_facility_count_data %>% 
  pivot_longer(
    cols = starts_with("malaria_")
    ) %>% 
  mutate(name = factor(name, levels = c("malaria_rdt_0-4", "malaria_rdt_5-14", "malaria_rdt_15", "malaria_tot"))) %>% 
  filter(name != "malaria_tot") %>% 
  ggplot(aes(x = name, y = value)) + 
  geom_col() + 
  labs(x = "age category",
